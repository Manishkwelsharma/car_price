pipeline {
   agent any
  
   environment {
       DOCKER_HUB_REPO = "sarutig/project-1"
       CONTAINER_NAME = "flask-hello-world"
 
   }
  
   stages {
       stage('Checkout') {
           steps {
               checkout([$class: 'GitSCM', branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/sarutig/flask-hello-world-devops-projectss.git']]])
           }
       }
       stage('Build') {
           steps {
               echo 'Building..'
               bat 'docker image build -t sarutig/project-1:latest .'
           }
       }
       stage('Test') {
           steps {
               echo 'Testing..'
               bat 'docker stop vigorous_bartik '
               bat 'docker rm vigorous_bartik '
               bat 'docker run --name vigorous_bartik sarutig/project-1 /bin/bash -c "pytest test.py && flake8"'
           }
       }
       stage('Deploy') {
           steps {
               echo 'Deploying....'
               bat 'docker stop vigorous_bartik'
               bat 'docker rm vigorous_bartik'
               bat 'docker run -d -p 5000:5000 --name vigorous_bartik sarutig/project-1'
           }
       }
   }
}
